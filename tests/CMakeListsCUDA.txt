cmake_minimum_required(VERSION 3.18 FATAL_ERROR)
project(tensor_test)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_CUDA_ARCHITECTURES "86")
enable_language(CUDA)
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/googletest)
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/cccl)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/cccl/thrust)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/cccl/libcudacxx/include)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/cccl/cub)
add_executable(${PROJECT_NAME} tests/test_tensor.cpp src/operations.cu)
target_link_libraries(${PROJECT_NAME} gtest gtest_main)
target_compile_options(${PROJECT_NAME} PRIVATE $<$<COMPILE_LANGUAGE:CXX>:-Wall -Wextra -Wpedantic>)
target_compile_options(${PROJECT_NAME} PRIVATE $<$<COMPILE_LANGUAGE:CUDA>:-Xcompiler -Wall,-Wextra,-Wpedantic,-Icccl/thrust,-Icccl/libcudacxx/include,-Icccl/cub>)
configure_file(src/tensor.hpp tensor.hpp COPYONLY)
